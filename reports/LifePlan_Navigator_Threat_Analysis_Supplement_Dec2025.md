### 7.2 SIEM検知ルール追加

```sql
-- MFAバイパス検知（トークンリプレイ）
SELECT user_id, session_token, COUNT(*) as replay_count
FROM auth_logs
WHERE event_type = 'token_validation'
  AND timestamp > NOW() - INTERVAL '1 hour'
GROUP BY user_id, session_token
HAVING COUNT(DISTINCT source_ip) > 1

-- API異常アクセス検知
SELECT api_endpoint, source_ip, COUNT(*) as request_count
FROM api_logs
WHERE timestamp > NOW() - INTERVAL '10 minutes'
GROUP BY api_endpoint, source_ip
HAVING COUNT(*) > 1000

-- ディープフェイク攻撃兆候（コールセンター）
SELECT caller_id, agent_id, call_duration
FROM call_center_logs
WHERE voice_verification_result = 'passed'
  AND subsequent_high_risk_action = true
  AND timestamp > NOW() - INTERVAL '24 hours'
```
