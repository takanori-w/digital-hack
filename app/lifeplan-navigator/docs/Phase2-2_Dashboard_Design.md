### 3.1 useLawRecommendations

```typescript
function useLawRecommendations(userProfile: UserProfile) {
  const [recommendations, setRecommendations] = useState<LawRecommendation[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    async function fetchRecommendations() {
      try {
        const response = await fetch('/api/laws?action=recommendations', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ userProfile }),
        });
        const data = await response.json();
        setRecommendations(data.recommendations);
      } catch (err) {
        setError(err as Error);
      } finally {
        setLoading(false);
      }
    }
    fetchRecommendations();
  }, [userProfile]);

  return { recommendations, loading, error };
}
```

### 3.2 usePersonalizedKeywords

```typescript
function usePersonalizedKeywords(userProfile: UserProfile) {
  const [keywords, setKeywords] = useState<string[]>([]);
  const [ageCategory, setAgeCategory] = useState<AgeCategory[]>([]);

  useEffect(() => {
    async function fetchKeywords() {
      const params = new URLSearchParams({
        action: 'personalized_keywords',
        employmentType: userProfile.employmentType,
        residenceType: userProfile.residenceType,
        householdType: userProfile.householdType,
        age: String(userProfile.age),
      });
      const response = await fetch(`/api/laws?${params}`);
      const data = await response.json();
      setKeywords(data.keywords);
      setAgeCategory(data.ageCategory);
    }
    fetchKeywords();
  }, [userProfile]);

  return { keywords, ageCategory };
}
```
